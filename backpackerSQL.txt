
CREATE TABLE Account (
  AccountId int NOT NULL AUTO_INCREMENT,
  name varchar (40) NOT NULL,
  username varchar (40) NOT NULL,
  password varchar(40) NOT NULL ,
  email varchar(40) NOT NULL,
  PRIMARY KEY (AccountId))
;

CREATE TABLE Gear (
  GearId int NOT NULL AUTO_INCREMENT,
  GearType varchar(40) NOT NULL,
  name varchar(40) NOT NULL,
  cost dec,
  rating int,
  PRIMARY KEY (GearId)
);

CREATE TABLE Location (
  LocationId int NOT NULL AUTO_INCREMENT ,
  name varchar(40) NOT NULL,
  description varchar(200), 
  PRIMARY KEY (LocationId)
);

CREATE TABLE Trip(
  TripId int NOT NULL AUTO_INCREMENT,
  name varchar(40) NOT NULL,
  description varchar(200),
  AccountId int,
  PRIMARY KEY (TripId),
  FOREIGN KEY AccountId REFERENCES Account(AccountId)
 
);

CREATE TABLE City (
  CityId int NOT NULL AUTO_INCREMENT,
  name varchar(40),
  PRIMARY KEY (CityId)
);

CREATE TABLE State (
 StateId int NOT NULL AUTO_INCREMENT, 
  name varchar(40),
  PRIMARY KEY (StateId)
);

CREATE TABLE Country (
  CountryId int NOT NULL AUTO_INCREMENT,
  name varchar(40),
  PRIMARY KEY (CountryId)
);

CREATE TABLE List (
  ListId int NOT NULL AUTO_INCREMENT,
  name varchar(40) NOT NULL,
  AccountId int,
  PRIMARY KEY (ListId),
  FOREIGN KEY AccountId REFERENCES Account(AccountId),
);

CREATE TABLE GearList (
GearListId int NOT NULL AUTO_INCREMENT,
  ListId int,
  GearId int,
  PRIMARY KEY (GearListId),
  FOREIGN KEY (ListId) REFERENCES List(ListId),
  FOREIGN KEY (GearId) REFERENCES Gear(GearId)
);

CREATE TABLE StateCountry (
StateCountryId int NOT NULL AUTO_INCREMENT,
  StateId int,
  CountryId int,
 PRIMARY KEY (StateCountryId),
 FOREIGN KEY (StateId) REFERENCES State(StateId),
 FOREIGN KEY (CountryId) REFERENCES Country(CountryId)
);

CREATE TABLE CityState (
CityStateId int NOT NULL AUTO_INCREMENT,
  StateId int,
  CityId int,
 PRIMARY KEY (CityStateId),
 FOREIGN KEY (CityId) REFERENCES City(CityId),
 FOREIGN KEY (StateId) REFERENCES State(StateId)
);


CREATE TABLE LocationCity (
LocationCityId int NOT NULL AUTO_INCREMENT,
  Location int, 
  CityId int,
PRIMARY KEY (LocationCityId),
FOREIGN KEY (CityId) REFERENCES City(CityId),
FOREIGN KEY (LocationId) REFERENCES Location(LocationId)

);

CREATE TABLE GearAccount (
GearAccountId int NOT NULL AUTO_INCREMENT,
  AccounId int,
  GearId int,
  PRIMARY KEY (GearAccountId),
  FOREIGN KEY (AccountId) REFERENCES Account(AccountId),
  FOREIGN KEY (GearId) REFERENCES Gear(GearId)
);

CREATE TABLE TripLocationList (
TripLocationListId int NOT NULL AUTO_INCREMENT,
  TripId int,
  LocationId, 
  PRIMARY KEY (TripLocationListId),
 FOREIGN KEY (TripId) REFERENCES Trip(TripId),
FOREIGN KEY (LocationId) REFERENCES Location(LocationID)

);
CREATE TABLE TripGearList(
TripGearListId int NOT NULL AUTO_INCREMENT,
  TripId int,
  ListId int,
  PRIMARY KEY (TripGearListId),
  FOREIGN KEY (TripId) REFERENCES Trip(TripId),
  FOREIGN KEY (ListId) REFERENCES List(ListId)
);


CREATE TABLE LocationAccount (
LocationAccountId int NOT NULL AUTO_INCREMENT,
  AccountId int,
  LocationId int,
  PRIMARY KEY (LocationAccountId),
  FOREIGN KEY (AccountId) REFERENCES Account(AccountId),
  FOREIGN KEY (LocationId) REFERENCES Location(LocationID)
);
